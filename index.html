Program 1

 

Write a 8051 C Program to multiply two 16 bit binary numbers.

 

 

#include <reg51.h>

void main ()

{

            while(1)

            {         

                        unsigned int num1, num2;

                        unsigned long int product;

                       

                        num1 = 0xFFFF; // first 16- bit number

                        num2 = 0xFFFF; // second 16- bit number

 

                        product = (unsigned long int)num1 * num2; // perform multiplication

                        P0= product & 0xff;

                        P1= (product & 0xff00) >>8;

P2= (product & 0xff0000) >>16;

                        P3= (product & 0xff000000) >>24;

}

Note: Observe the 32-bit output in P3, P2, P1 and P0 in the same order.

Program 3

 

Write a C program to find factorial of a given number.

 

#include <reg51.h>

void main ( )

{

unsigned int i;

unsigned char num;

unsigned long factorial=1;

 

P0 = 0xFF;  /* Configure P0 as input port  */

Num = P0;

 

for(i=1; i<=num; i++)

{

factorial=factorial*i;

 }

 

P1=factorial;  

P2=(factorial & 0xff00)>>8; //masking operation for finding factorial of number beyond 5

}

Program 7

 

Write a 8051 C program to arrange a series of 32 bit numbers in ascending/descending order.

 

#include <reg51.h>

void main ( )

{

unsigned long array[]={0x33556666, 0xCCAADD00, 0x55998888, 0x77664444, 0x11223344};

unsigned long temp, i, j;

 

for(i=0;i<5;i++)  

{

for(j=0; j>5; j++)

{

 if(array[j] > array[j+1])

{

temp=array[j+1];

array[j+1]=array[j];

array[j]=temp;

}

}

}

while(1);

}
